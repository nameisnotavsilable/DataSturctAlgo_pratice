//首次初始化仓库
# 1. 进入项目目录
cd my-project

# 2. 初始化本地仓库
git init

# 3. 配置用户信息（如果没配置过）
git config --global user.name "张三"
git config --global user.email "zhangsan@example.com"

# 4. 添加所有文件
git add .

# 5. 提交
git commit -m "初始提交"

# 6. 关联远程仓库
git remote add origin https://gitee.com/zhangsan/my-project.git

# 7. 推送（注意分支名可能是main或master）
# 查看当前分支
git branch
# 如果显示master，重命名为main
git branch -M main

# 8. 推送
git push -u origin main
//更新仓库
# 1. 查看状态
git status

# 2. 添加更改的文件
git add .

# 3. 提交
git commit -m "修复了登录bug"

# 4. 拉取远程更新（避免冲突）
git pull origin main

# 5. 推送
git push origin main
